1.Find the Smaller Angle

        
var Minimal_Angle = (h, m) => 
{
    
  let hourAngle = (360 / 12) * (h + (m/ 60));
  
  let minuteAngle = (360 / 60) * m;
  
  let angle = Math.abs(hourAngle - minuteAngle);
  return Math.min(angle, 360 - angle);   
};

        var Perfect_Check = (N) => 
{
     let sum = 1;
    for (let i = 1; i < N; i++) {
        if (N % i === 0) {
            sum += i;
        }
    }
if(sum === N) {
  return "YES"
}
else
{
  return "NO"
}
};

4.Reverse a Number.

        var Reverse_Number = (N) => 
        {
         
        let reversedNum = 0;
        while (N> 0) {
            reversedNum = reversedNum * 10 + (N % 10);
            N = Math.floor(N / 10);
        }
        return reversedNum;
          
        
        };
